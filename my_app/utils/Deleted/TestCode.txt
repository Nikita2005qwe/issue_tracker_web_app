# Много импорта
from typing import Optional
from authorization.services import AuthorizationContext
from issue_tracker.services import IssueTrackerContext


# Большой словарь  повторяющимся текстом
dict_of_name_functions: dict[str, callable] = {
    "AuthorizationContext.get_context_for_index_page": AuthorizationContext.get_context_for_index_page,
    "IssueTrackerContext.get_context_for_index_page": IssueTrackerContext.get_context_for_index_page,
    "IssueTrackerContext.get_context_after_delete": IssueTrackerContext.get_context_after_delete,
    "IssueTrackerContext.get_context_after_add_task": IssueTrackerContext.get_context_after_add_task,
}

# Слабая функцональность у класса
class FunctionsUtils:
    """
    Класс обрабатывающий объекты функций
    """

    @staticmethod
    def get_function_by_name(function_name: Optional[str]):
        """
        Динамический импорт функции

        Обязательно в файле конфигурации указывать
        Класс функции и название функции через точку
        :param function_name: Имя функции.
        :return: Объект функции.
        """
        if function_name is not None:
            return dict_of_name_functions[function_name]

        return None
